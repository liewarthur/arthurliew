    <script>
    /* --- SCROLL NAVIGATION HIGHLIGHTS --- */
    const navLinks = document.querySelectorAll('nav a');
    const sections = document.querySelectorAll('section');
    window.addEventListener('scroll', () => {
    let current = '';
    sections.forEach(section => {
    const sectionTop = section.offsetTop;
    if (pageYOffset >= sectionTop - 150) {
    current = section.getAttribute('id');
    }
    });
    navLinks.forEach(link => {
    link.classList.remove('active');
    if (link.getAttribute('href').includes(current)) {
    link.classList.add('active');
    }
    });
    });
    /* --- SCALABLE MODAL DISPLAY ENGINE --- */
    const bodyElement = document.body;
    let activeGalleryIndex = 0;
    let currentActiveGallery = [];
    let currentImgElement = null;
    // Open Modal Controller
    document.querySelectorAll('.card[data-modal-target]').forEach(card => {
    card.addEventListener('click', () => {
    const targetId = card.getAttribute('data-modal-target');
    const modal = document.getElementById(targetId);
    if (modal) {
    modal.classList.add('active');
    bodyElement.classList.add('modal-open');
    // Parse custom inline slide list array from HTML data attribute target
    currentImgElement = modal.querySelector('.gallery-main-img');
    if (currentImgElement && currentImgElement.hasAttribute('data-gallery')) {
    currentActiveGallery = currentImgElement.getAttribute('data-gallery').split(',');
    } else {
    currentActiveGallery = [currentImgElement.src];
    }
    activeGalleryIndex = 0;
    updateGalleryImage();
    }
    });
    });
    // Global Dynamic Gallery Image Switcher Renderer Engine Loop
    function updateGalleryImage() {
    if (currentImgElement && currentActiveGallery.length > 0) {
    currentImgElement.src = currentActiveGallery[activeGalleryIndex];
    }
    }
    // Global Modal Event Listeners (Close Toggles / Nav Arrows inside wrapper bounds)
    document.querySelectorAll('.modal').forEach(modal => {
    // Close Button Event Toggles
    modal.querySelector('.modal-close').addEventListener('click', () => {
    closeModal(modal);
    });
    // Close modal when hitting backdrop area outside container card box
    modal.addEventListener('click', (e) => {
    if (e.target === modal) {
    closeModal(modal);
    }
    });
    // Next Arrow Action Button Hook
    const nextBtn = modal.querySelector('.gallery-nav.next');
    if (nextBtn) {
    nextBtn.addEventListener('click', (e) => {
    e.stopPropagation();
    if (currentActiveGallery.length > 0) {
    activeGalleryIndex = (activeGalleryIndex + 1) % currentActiveGallery.length;
    updateGalleryImage();
    }
    });
    }
    // Prev Arrow Action Button Hook
    const prevBtn = modal.querySelector('.gallery-nav.prev');
    if (prevBtn) {
    prevBtn.addEventListener('click', (e) => {
    e.stopPropagation();
    if (currentActiveGallery.length > 0) {
    activeGalleryIndex = (activeGalleryIndex - 1 + currentActiveGallery.length) % currentActiveGallery.length;
    updateGalleryImage();
    }
    });
    }
    });
    function closeModal(modalInstance) {
    modalInstance.classList.remove('active');
    bodyElement.classList.remove('modal-open');
    }
    // Close on escape key configuration rule
    window.addEventListener('keydown', (e) => {
    if (e.key === 'Escape') {
    const openModal = document.querySelector('.modal.active');
    if (openModal) closeModal(openModal);
    }
    });
    </script>
